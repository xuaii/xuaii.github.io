def findOptDiv(root):
    losses = []
    for attr in root.attrList:
        for s in root.Dataset[attr]:
            losses.append((loss(attr, s, root.Dataset), attr, s))
    minLoss = min(losses)
    return minLoss

def predict(x, root):
    while(len(root.attrList) != 0):
